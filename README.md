The Restful server runs on Azure

# API
'info': 'GET /api/v1'

'statistics': 'GET /api/v1/general'

'countries infected': 'GET /api/v1/countries'

'country stats': 'GET /api/v1/stats?name='

'TODO: latest news': 'GET /api/v1/news'
